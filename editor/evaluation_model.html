def evaluate_safety(telemetry_data):
risk_score = 0
# 计算风力影响
if telemetry_data.wind_speed > 12:
risk_score += (wind_speed - 12) * 1.5
# 电量临界值校验
if telemetry_data.battery < 15:
risk_score += 20
return min(100, risk_score)
